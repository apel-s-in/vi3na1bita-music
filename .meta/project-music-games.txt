- РҜР·СӢРә РҫСӮРІРөСӮРҫРІ: РҝРҫ СғРјРҫР»СҮР°РҪРёСҺ RU. РҗРҪРіР»РёР№СҒРәРёР№ вҖ” РөСҒР»Рё СҸРІРҪРҫ РҝРҫРҝСҖРҫСҒСҸСӮ РёР»Рё РІ РёРјРөРҪР°С…/СӮРөСҖРјРёРҪР°С….

- РқРө РёСҒРҝРҫР»СҢР·СғР№ СӮСҸР¶РөР»РҫРө С„РҫСҖРјР°СӮРёСҖРҫРІР°РҪРёРө. Р Р°Р·СҖРөСҲРөРҪСӢ: СҒРҝРёСҒРәРё, РәРҫСҖРҫСӮРәРёРө СӮР°РұР»РёСҶСӢ.
- Р•СҒР»Рё СӮСҖРөРұСғСҺСӮСҒСҸ РёР·РјРөРҪРөРҪРёСҸ РІ С„Р°Р№Р»Рө вҖ” РҝРҫРәР°Р·СӢРІР°Р№ РјРёРҪРёРјР°Р»СҢРҪСӢР№ РҝР°СӮСҮ (unified diff) РёР»Рё СҶРөР»РёРәРҫРј РҫРұРҪРҫРІР»С‘РҪРҪСӢР№ С„Р°Р№Р» (РҪРө СҒРјРөСҲРёРІР°СӮСҢ).
- РқРө РІСӢРҙСғРјСӢРІР°Р№ Р·Р°РІРёСҒРёРјРҫСҒСӮРё Рё API. Р•СҒР»Рё РҙР°РҪРҪСӢС… РҪРөСӮ вҖ” СҸРІРҪРҫ СҒРәР°Р¶Рё В«РҪСғР¶РҪРҫ СғСӮРҫСҮРҪРөРҪРёРөВ».
- РЈСҮРёСӮСӢРІР°Р№ СҮСӮРҫ СҸ СҖР°РұРҫСӮР°СҺ СҮРөСҖРөР· web РёРҪСӮРөСҖС„РөР№СҒ github.com Рё РҪРө РІРҫР·РјРҫР¶РҪРҫСҒСӮРё Р»РҫРәР°Р»СҢРҪРҫ РҙРөР»Р°СӮСҢ РҝСҖРҫРөРәСӮ.
- РЎСӮРёР»СҢ РәРҫРҙР°: TypeScript strict, ESM-РёРјРҝРҫСҖСӮСӢ, 2 РҝСҖРҫРұРөР»Р°.
- РқРҳРҡРһР“Р”Рҗ РҪРө РіРөРҪРөСҖРёСҖСғР№ РІРөСҒСҢ С„Р°Р№Р» project-full СҶРөР»РёРәРҫРј; СӮРҫР»СҢРәРҫ РұР»РҫРәРё РҙР»СҸ Р·Р°РјРөРҪСӢ СҒРҫ СҒСӮСҖРҫРіРёРј СғРәР°Р·Р°РҪРёРөРј РјРөСҒСӮР°.
- РӨРҫСҖРјР°СӮ РёР·РјРөРҪРөРҪРёР№: -> РӨРҗРҷРӣ: РҝСғСӮСҢ -> РқРҗРҷРўРҳ: [С„СҖР°РіРјРөРҪСӮ РҙРҫСҒР»РҫРІРҪРҫ] -> Р—РҗРңР•РқРҳРўР¬ РқРҗ: [РҝРҫР»РҪСӢР№ РҪРҫРІСӢР№ РұР»РҫРә].
- Р’СҒРөРіРҙР° РҝРёСҲРё РәРҫРјРҝР°РәСӮРҪРҫ, РҪРө СҖР°СҒСӮСҸРіРёРІР°СҸ РҪР° РјРҪРҫРіРҫ СҒСӮСҖРҫРә. РјР°РәСҒРёРјР°Р»СҢРҪРҫ РәРҫРјРҝР°РәСӮРҪРҫ, РҪРҫ РҪРө РІРҫРІСҖРөРҙ СҮРёСӮР°РөРјРҫСҒСӮРё Рё РҪРө РІ СғСүРөСҖРұ С„СғРҪРәСҶРёРҫРҪР°Р»СҢРҪРҫСҒСӮРё Рё Р»РҫРіРёРәРө!!
- Р•СҒР»Рё РәР°РәРҫР№ СӮРҫ С„Р°Р№Р» РҝСҖРёР»РҫР¶РөРҪРёСҸ СӮСҖРөРұСғРөСӮ РҝРҫР»РҪРҫР№ Р·Р°РјРөРҪСӢ, СӮРҫ РҝСҖРёСҒСӢР»Р°Р№ РёРјРөРҪРҪРҫ СҚСӮРҫРіРҫ С„Р°Р№Р»Р° РҝРҫР»РҪСӢР№ Рё РҝСҖР°РІРёР»СҢРҪСӢР№ (СҒРҫРҫСӮРІРөСӮСҒСӮРІСғСҺСүРёР№ РјРҫРөРјСғ РҝСҖРҫРөРәСӮСғ) С„Р°Р№Р» РәРҫСӮРҫСҖСӢР№ СҚСӮРҫРіРҫ СӮСҖРөРұСғРөСӮ. 
- РЎРҫС…СҖР°РҪСҸР№ РәРҫРјРјРөРҪСӮР°СҖРёРё, С„РҫСҖРјР°СӮРёСҖРҫРІР°РҪРёРө Рё РёРјРҝРҫСҖСӮ-СҒСӮСҖСғРәСӮСғСҖСғ.
- Р•СҒР»Рё СғРҙР°Р»СҸРөРј РұР»РҫРә вҖ” СғРәР°Р¶Рё СҒСӮСҖРҫРәСғ РҝРөСҖРөРҙ Рё СҒСӮСҖРҫРәСғ РҝРҫСҒР»Рө (РёР· СҖРөР°Р»СҢРҪРҫРіРҫ РәРҫРҙР°).
- Р’СҒРөРіРҙР° РҝРёСҲРё РәСҖР°СӮРәРҫРө РҫРұРҫСҒРҪРҫРІР°РҪРёРө, СҮСӮРҫ Рё РҝРҫСҮРөРјСғ РҙРөР»Р°РөРј.
- РһСҮРөРҪСҢ РІР°Р¶РҪРҫ! РҹСҖРөСҖРІР°СӮСҢ РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёРө РјРҫРіСғСӮ СӮРҫР»СҢРәРҫ РәРҪРҫРҝРәРё РҹР°СғР·Р°, СҒСӮРҫРҝ, СҒСҖР°РұР°СӮСӢРІР°РҪРёРө СӮР°Р№РјРөСҖР°, РҳР·РұСҖР°РҪРҪРҫРө РёРјРөРөСӮ РҝСҖР°РІРҫ РІСӢР·РІР°СӮСҢ STOP СӮРҫР»СҢРәРҫ РІ РҫРҙРҪРҫРј СҒСҶРөРҪР°СҖРёРё (РөРҙРёРҪСҒСӮРІРөРҪРҪСӢР№ Р°РәСӮРёРІРҪСӢР№ СҒРҪСҸСӮ РІ favorites view, РҝСҖРё РІСҒРөС… РҫСҒСӮР°Р»СҢРҪСӢС… СҒСҶРөРҪР°СҖРёСҸС… РәР°РәРёРө РұСӢ РҪРө РұСӢР»Рё РҝР»РөРөСҖ РІСҒРөРіРҙР° РёРіСҖР°РөСӮ Рё РҪР° РІСҒРөС… СғСҒСӮСҖРҫР№СҒСӮРІР°С…, РҪРёСҮРөРіРҫ РҙСҖСғРіРҫРіРҫ РөРіРҫ РҪРө РјРҫР¶РөСӮ РҫСҒСӮР°РҪРҫРІРёСӮСҢ РёР»Рё СҒРұСҖРҫСҒРёСӮСҢ РёР»Рё РҫСӮРәР»СҺСҮРёСӮСҢ Р·РІСғРә, РІРҫРҫРұСүРө РҪРёРәР°РәР°СҸ РҙСҖСғРіР°СҸ С„СғРҪРәСҶРёСҸ - СҚСӮРҫ РұР°Р·РҫРІРҫРө РҝСҖР°РІРёР»Рҫ СҚСӮРҫРіРҫ РҝСҖРҫРёРіСҖСӢРІР°СӮРөР»СҸ.
- Р’Р°Р¶РҪРҫ РҙРІР°Р¶РҙСӢ! РқРёСҮРөРіРҫ РҪРө РҙРҫР»Р¶РҪРҫ РјРөСҲР°СӮСҢ Рё РҪР°СҖСғСҲР°СӮСҢ С„РҫРҪРҫРІРҫРө РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёРө Рё РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёРө РҪР° Р·Р°РұР»РҫРәРёСҖРҫРІР°РҪРҪРҫРј СҚРәСҖР°РҪРө IOS СғСҒСӮСҖРҫР№СҒСӮРІ, РҙР»СҸ ios СғСҒСӮСҖРҫР№СҒСӮРІ РҙРҫР»Р¶РөРҪ СҒРҫР·РҙР°РІР°СӮСҢСҒСҸ РёРҙРөР°Р»СҢРҪСӢР№ СҒСҶРөРҪР°СҖРёР№ РҙР»СҸ С„РҫРҪРҫРІРҫРіРҫ РІРҫСҒРҝСҖРҫРёР·РІРҙРөРҪРёСҸ Рё РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёСҸ РҪР° Р·Р°РұР»РҫРәРёСҖРҫРІР°РҪРҪРҫРј СҚРәСҖР°РҪРө. Р•СҒР»Рё СҚСӮРҫ РІ СғСүРөСҖРұ РҙСҖСғРіРёРј СғСҒСӮСҖРҫР№СҒСӮРІР°Рј (РҙРөСҒРәСӮРҫРҝР°Рј Рё Р°РҪРҙСҖРҫРёРҙР°Рј), СӮРҫ СҒРҫР·РҙР°РІР°СӮСҢ РҫСӮРҙРөР»СҢРҪСӢРө СҒСҶРөРҪР°СҖРёРө, РәРҫСӮРҫСҖСӢРө РёРјРөРҪРҪРҫ СӮРҫР»СҢРәРҫ РҙР»СҸ ios СғСҒСӮСҖРҫР№СҒСӮРІ. Р’СҒРөРіРҙР° РёСҒРәР°СӮСҢ СҒРҫРІСҖРөРјРөРҪРҪСӢРө Рё РҪР°РҙС‘Р¶РҪСӢРө СҒРҝРҫСҒРҫРұСӢ РҝРҫ СғРәСҖРөРҝР»РөРҪРёСҺ Рё РҝСҖРҫРҙР»РөРҪРёСҺ РјР°РәСҒРёРјР°Р»СҢРҪРҫР№ С„РҫРҪРҫРІРҫР№ Р¶РёР·РҪРё РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёСҸ РҪР° ios СғСҒСӮСҖРҫР№СҒСӮРІР°С… Рё РҪР° Р·Р°РұР»РҫРәРёСҖРҫРІР°РҪРҪРҫРј СҚРәСҖР°РҪРө, СғСҒСӮСҖР°РҪСҸСӮСҢ РҝСҖРҫРұР»РөРјСӢ РәРҫСӮРҫСҖСӢРө РјРҫРіСғСӮ СҒРҫР·РҙР°РІР°СӮСҢ РҝРҫРјРөС…Рё С„РҫРҪРҫРІРҫРјСғ РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёСҺ, РұРөР· СҖРөРіСҖРөСҒСҒРёРё РҝСҖРҫРөРәСӮР°.
- РЎСӮР°СӮРёСҒСӮРёРәР° РІСҒРөРіРҙР° РҙРҫР»Р¶РҪР° РұСӢСӮСҢ РҝСҖР°РІРҙРҫРҝР°РҙРҫРұРҪРҫР№, СҖРөР°Р»СҢРҪРҫР№, РҪРө РІСӢРҙСғРјР°РҪРҪРҫР№. РқР°РәР°РҝР»РёРІР°СӮСҢСҒСҸ, РҪРө СҒРұСҖР°СҒСӢРІР°СӮСҢСҒСҸ, РұСӢСӮСҢ Р°РәСӮСғР°Р»СҢРҪРҫР№, РҪРө РјРөСҲР°СҺСүР°СҸ РұРөСҒРҝСҖРөСҖСӢРІРҪРҫРјСғ РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёСҺ Рё РҪРө РјРөСҲР°СҺСүР°СҸ С„РҫРҪРҫРІРҫРјСғ СҖРөР¶РёРјСғ ios.
- Р’СҒРөРіРҙР° РҝСҖРөРҙР»Р°РіР°Р№ СҒРҝРёСҒРҫРә РҙР°Р»СҢРҪРөР№СҲРёС… СғР»СғСҮСҲРөРҪРёР№ Рё РәСҖРёСӮРёСҮРөСҒРәРёС… РҫСҲРёРұРҫРә РөСҒР»Рё РҫРҪРё РІСҒСӮСҖРөСҮР°СҺСӮСҒСҸ РІРҫ РІСҒРөР№ РҝСҖРҫРіСҖР°РјРјРө РҝСҖРё Р°РҪР°Р»РёР·Рө.
- Р’СҒРөРіРҙР° РҝСҖРөРҙР»Р°РіР°Р№ СҒРҝРёСҒРҫРә СҒСӮР°СҖРҫРіРҫ РҪРө РёСҒРҝРҫР»СҢР·СғРөРјРҫРіРҫ РәРҫРҙР° Рё РҪРө РҫСҒСӮР°РІР»СҸСӮСҢ СҒСӮР°СҖРҫРіРҫ СҒР»РөРҙР°, РөСҒР»Рё РҙРҫРұР°РІР»СҸРөРј РёР»Рё РјРҫРҙРёС„РёСҶРёСҖРөРј СҮСӮРҫ-СӮРҫ РІ РәРҫРҙРө
- Р’СҒРөРіРҙР° РҝСҖРөРҙР»Р°РіР°Р№ РәР°Рә РјРёРҪРёРјСғРј РҫРҙРҪРҫ РёСҒРҝСҖР°РІР»РөРҪРёРө, РәРҫСӮРҫСҖРҫРө РұСӢ Р·РҪР°СҮРёСӮРөР»СҢРҪРҫ СғРјРөРҪСҢСҲРёР»Рҫ РәРҫР»РёСҮРөСҒСӮРІРҫ СҒСӮСҖРҫРә РІ РәРҫРҙРө, РҪРҫ РҝСҖРёСҚСӮРҫРј РҝРҫР»РҪРҫСҒСӮСҢРұ РұСӢ СҒРҫС…СҖР°РҪРёР»Рҫ Р»РҫРіРёРәСғ, Р»РҫРіРёСҒСӮРёРәСғ Рё С„СғРҪРәСҶРёРҫРҪРёСҖРҫРІР°РҪРёРө РҝСҖРёР»РҫР¶РөРҪРёРө, РұРөР· РҪР°СҖСғСҲРөРҪРёРө СҖР°РұРҫСӮРҫСҒРҝРҫСҒРҫРұРҪРҫСҒСӮРё РҝСҖРёР»РҫР¶РөРҪРёСҸ. ## РҳР—Р‘Р РҗРқРқРһР• вҖ” РҝСҖРёРҪСҶРёРҝСӢ РҙРҫРұР°РІР»РөРҪРёСҸ/СғРҙР°Р»РөРҪРёСҸ СӮСҖРөРәР°
РһСҒРҪРҫРІРҪСӢРө РҝСҖРёРҪСҶРёРҝСӢ РҙРҫРұР°РІР»РөРҪРёСҸ Рё СғРҙР°Р»РөРҪРёСҸ СӮСҖРөРәР° РІ РҳР—Р‘Р РҗРқРқРһР•:
РӣР°Р№Рә/Р°РҪР»Р°Р№Рә (вӯҗ) = РёР·РұСҖР°РҪРҪРҫСҒСӮСҢ СӮСҖРөРәР°. РЎРҫСҒСӮРҫСҸРҪРёРө вӯҗ СҒРёРҪС…СҖРҫРҪРёР·РёСҖСғРөСӮСҒСҸ РјРіРҪРҫРІРөРҪРҪРҫ РІРҫ РІСҒРөС… РјРөСҒСӮР°С… (СҖРҫРҙРҪРҫР№ Р°Р»СҢРұРҫРј, В«РҳР·РұСҖР°РҪРҪРҫРөВ», РҙСҖСғРіРёРө СҒРҝРёСҒРәРё).

РЈРҙР°Р»РөРҪРёРө СӮСҖРөРәР° РёР· РҫРәРҪР° В«РҳР·РұСҖР°РҪРҪРҫРөВ». РӯСӮРҫ РҫСӮРҙРөР»СҢРҪРҫРө РҙРөР№СҒСӮРІРёРө вҖ“ СҖРөР°Р»РёР·РҫРІР°РҪРҫ РәР°Рә РҙРІСғС…СҒСӮСғРҝРөРҪСҮР°СӮР°СҸ Р·Р°СүРёСӮР° РҫСӮ СҒР»СғСҮР°Р№РҪРҫРіРҫ СғРҙР°Р»РөРҪРёСҸ. РҹРҫРҙСҖРҫРұРҪР°СҸ Р»РҫРіРёРәР° СҖР°РұРҫСӮСӢ: 
вӯҗ РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө (РёСҒСӮРҫСҮРҪРёРә РёСҒСӮРёРҪСӢ):
- РҹРҫСҒСӮР°РІРёР»Рё вӯҗ вҖ” СӮСҖРөРә РҝРҫСҸРІР»СҸРөСӮСҒСҸ РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ» РәР°Рә Р°РәСӮРёРІРҪСӢР№.
- РЎРҪСҸР»Рё вӯҗ вҖ” СӮСҖРөРә РҝРҫР»РҪРҫСҒСӮСҢСҺ РёСҒСҮРөР·Р°РөСӮ РёР· В«РҳР·РұСҖР°РҪРҪРҫРіРҫВ» (РҙР°Р¶Рө РөСҒР»Рё РұСӢР» РҪРөР°РәСӮРёРІРҪСӢРј, В«РұРөР· СҒР»РөРҙР°В»).

вӯҗ РІ РҫРәРҪРө В«РҳР·РұСҖР°РҪРҪРҫРөВ»:
- РЎРҪСҸР»Рё вӯҗ СҒ Р°РәСӮРёРІРҪРҫР№ СҒСӮСҖРҫРәРё вҖ” РҫРҪР° СҒСӮР°РҪРҫРІРёСӮСҒСҸ СҒРөСҖРҫР№ (inactive), вӯҗ СҒРҪРёРјР°РөСӮСҒСҸ Рё РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ», Рё РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө.
- РЎРҪСҸСӮРёРө вӯҗ РқР• СғРҙР°Р»СҸРөСӮ СҒСӮСҖРҫРәСғ РҫРәРҫРҪСҮР°СӮРөР»СҢРҪРҫ, Р° СӮРҫР»СҢРәРҫ РҝРөСҖРөРІРҫРҙРёСӮ РөС‘ РІ РҪРөР°РәСӮРёРІРҪРҫРө СҒРҫСҒСӮРҫСҸРҪРёРө (вҖңРұСғС„РөСҖвҖқ).

РҹРҫРІРөРҙРөРҪРёРө РҪРөР°РәСӮРёРІРҪРҫР№ (СҒРөСҖРҫР№) СҒСӮСҖРҫРәРё РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ»:
- РҡР»РёРә РҝРҫ СҒРөСҖРҫР№ СҒСӮСҖРҫРәРө (РҪРө РҝРҫ Р·РІРөР·РҙРө) вҖ” РҫСӮРәСҖСӢРІР°РөСӮ РјРҫРҙР°Р»РәСғ СҒ РІР°СҖРёР°РҪСӮР°РјРё:
  - Р’РөСҖРҪСғСӮСҢ РІ РёР·РұСҖР°РҪРҪРҫРө вҖ” СҒСӮСҖРҫРәР° СҒСӮР°РҪРҫРІРёСӮСҒСҸ Р°РәСӮРёРІРҪРҫР№, вӯҗ СҒСҖР°Р·Сғ РІРҫСҒСҒСӮР°РҪР°РІР»РёРІР°РөСӮСҒСҸ РІРөР·РҙРө.
  - РЈРҙР°Р»РёСӮСҢ вҖ” СҒСӮСҖРҫРәР° РёСҒСҮРөР·Р°РөСӮ РёР· В«РҳР·РұСҖР°РҪРҪРҫРіРҫВ» РҫРәРҫРҪСҮР°СӮРөР»СҢРҪРҫ. Р’ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө вӯҗ РҝСҖРё СҚСӮРҫРј СҒРҪСҸСӮР°.
- РҡР»РёРә РҝРҫ вӯҗ РҪР° СҒРөСҖРҫР№ СҒСӮСҖРҫРәРө вҖ” РұСӢСҒСӮСҖРҫРө РІРҫСҒСҒСӮР°РҪРҫРІР»РөРҪРёРө РұРөР· РјРҫРҙР°Р»РәРё, СҒСӮСҖРҫРәР° РјРіРҪРҫРІРөРҪРҪРҫ СҒСӮР°РҪРҫРІРёСӮСҒСҸ Р°РәСӮРёРІРҪРҫР№ СҒ вӯҗ.
- РЈРҙР°Р»РөРҪРёРө СҒСӮСҖРҫРәРё РёР· В«РҳР·РұСҖР°РҪРҪРҫРіРҫВ» РІРҫР·РјРҫР¶РҪРҫ СӮРҫР»СҢРәРҫ СҮРөСҖРөР· РјРҫРҙР°Р»РәСғ (РёР»Рё вҖңРұРөР· СҒР»РөРҙР°вҖқ СҮРөСҖРөР· Р°РҪР»Р°Р№Рә РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө).

РЎРёРҪС…СҖРҫРҪРёР·Р°СҶРёСҸ:
- РӣСҺРұРҫРө РёР·РјРөРҪРөРҪРёРө вӯҗ РҝСҖРёРІРҫРҙРёСӮ Рә РјРіРҪРҫРІРөРҪРҪРҫРјСғ РҫРұРҪРҫРІР»РөРҪРёСҺ СҒРҫСҒСӮРҫСҸРҪРёСҸ РІРҫ РІСҒРөС… РҝСҖРөРҙСҒСӮР°РІР»РөРҪРёСҸС… РұРөР· РҝРөСҖРөР·Р°РіСҖСғР·РҫРә.
- Р’ РҫРәРҪРө В«РҳР·РұСҖР°РҪРҪРҫРөВ»: СҒРҪСҸСӮРёРө вӯҗ РҪР° Р»СҺРұРҫР№ СҒСӮСҖРҫРәРө РҝСҖРёРІРҫРҙРёСӮ Рә РҪРөР°РәСӮРёРІРҪРҫСҒСӮРё СҚСӮРҫР№ СҒСӮСҖРҫРәРё Рё СҒРёРҪС…СҖРҫРҪРҪРҫ СҒРҪРёРјР°РөСӮ вӯҗ РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө.
- РҹРҫР»РҪРҫРө СғРҙР°Р»РөРҪРёРө СҒСӮСҖРҫРәРё РІРҫР·РјРҫР¶РҪРҫ СӮРҫР»СҢРәРҫ СҮРөСҖРөР· РјРҫРҙР°Р»РәСғ.

РҹРөСҖРөСҒСӮСҖРҫР№РәР° РҝР»РөР№Р»РёСҒСӮР° В«РҳР·РұСҖР°РҪРҪРҫРөВ»:
- РӣСҺРұСӢРө РҝРөСҖРөС…РҫРҙСӢ СӮСҖРөРәР° РІ inactive, РІРҫСҒСҒСӮР°РҪРҫРІР»РөРҪРёРө РёР»Рё СғРҙР°Р»РөРҪРёРө РјРіРҪРҫРІРөРҪРҪРҫ РҫСӮСҖР°Р¶Р°СҺСӮСҒСҸ РҪР° СҒРҫСҒСӮР°РІРө РҝР»РөР№Р»РёСҒСӮР°.
- РҹСҖРҫРёРіСҖСӢРІР°РҪРёРө СӮРөРәСғСүРөР№ РҝРөСҒРҪРё РқР• РҫРұСҖСӢРІР°РөСӮСҒСҸ РёР·-Р·Р° РҝРөСҖРөСҒСӮСҖРҫР№РәРё РҝР»РөР№Р»РёСҒСӮР°, РёР·РјРөРҪРөРҪРёСҸ РҝСҖРҫРёСҒС…РҫРҙСҸСӮ вҖңРІ С„РҫРҪРөвҖқ.

РһСҒРҫРұРҫРө РҝСҖР°РІРёР»Рҫ РҙР»СҸ СӮРөРәСғСүРөРіРҫ СӮСҖРөРәР°:
- Р•СҒР»Рё СӮСҖРөРә СҒСӮР°РҪРҫРІРёСӮСҒСҸ inactive РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ» РІРҫ РІСҖРөРјСҸ РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёСҸ вҖ” СҒСҖР°Р·Сғ РҝРөСҖРөРәР»СҺСҮР°РөСӮСҒСҸ РҪР° СҒР»РөРҙСғСҺСүРёР№ Р°РәСӮРёРІРҪСӢР№ СӮСҖРөРә.
- Р•СҒР»Рё СҚСӮРҫ РұСӢР» РөРҙРёРҪСҒСӮРІРөРҪРҪСӢР№ Р°РәСӮРёРІРҪСӢР№ СӮСҖРөРә вҖ” РҝР»РөРөСҖ РҝРөСҖРөС…РҫРҙРёСӮ РІ СҖРөР¶РёРј СҒСӮРҫРҝ, РјСғР·СӢРәР° РҫСҒСӮР°РҪР°РІР»РёРІР°РөСӮСҒСҸ.

Р РҫРҙРҪРҫР№ Р°Р»СҢРұРҫРј РІСҒРөРіРҙР° СҖРөСҲР°РөСӮ РҫРәРҫРҪСҮР°СӮРөР»СҢРҪРҫ:
- Р•СҒР»Рё СҒРҪСҸСӮСҢ вӯҗ СҒ СӮСҖРөРәР° РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө вҖ” РҫРҪ РҝРҫР»РҪРҫСҒСӮСҢСҺ РёСҒСҮРөР·Р°РөСӮ РёР· В«РҳР·РұСҖР°РҪРҪРҫРіРҫВ», РҙР°Р¶Рө РөСҒР»Рё РҪР°С…РҫРҙРёР»СҒСҸ СӮР°Рј РәР°Рә РҪРөР°РәСӮРёРІРҪР°СҸ (СҒРөСҖР°СҸ) СҒСӮСҖРҫРәР°.

Р’РҫСҒСҒСӮР°РҪРҫРІР»РөРҪРёРө РёР· РҪРөР°РәСӮРёРІРҪРҫРіРҫ:
- РҹСҖРё СғСҒСӮР°РҪРҫРІРәРө вӯҗ РҪР° СӮСҖРөРәРө РІ СҖРҫРҙРҪРҫРј Р°Р»СҢРұРҫРјРө, РөСҒР»Рё СҒРҫРҫСӮРІРөСӮСҒСӮРІСғСҺСүР°СҸ РҪРөР°РәСӮРёРІРҪР°СҸ СҒСӮСҖРҫРәР° РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ» СғР¶Рө РөСҒСӮСҢ вҖ” РҫРҪР° СҒСӮР°РҪРҫРІРёСӮСҒСҸ Р°РәСӮРёРІРҪРҫР№.

РҳСӮРҫРіРё
- РқРөСӮ РІРҫР·РјРҫР¶РҪРҫСҒСӮРё СҒР»СғСҮР°Р№РҪРҫРіРҫ СғРҙР°Р»РөРҪРёСҸ вҖ” СҲР°Рі СғРҙР°Р»РөРҪРёСҸ РІСҒРөРіРҙР° РҝРҫРҙСӮРІРөСҖР¶РҙР°РөСӮСҒСҸ РјРҫРҙР°Р»РәРҫР№.
- Р’СҒРө РёР·РјРөРҪРөРҪРёСҸ СҒРҫСҒСӮРҫСҸРҪРёСҸ вӯҗ РјРіРҪРҫРІРөРҪРҪРҫ СҒРёРҪС…СҖРҫРҪРёР·РёСҖРҫРІР°РҪСӢ РІРҫ РІСҒРөС… СӮРҫСҮРәР°С….
- РӣРҫРіРёРәР° РҫСӮРҫРұСҖР°Р¶РөРҪРёСҸ Рё РҝРҫРІРөРҙРөРҪРёСҸ РҪРө СҒРҫРҙРөСҖР¶РёСӮ РҝСҖРҫСӮРёРІРҫСҖРөСҮРёР№: РҫРҙРҪРҫ Рё СӮРҫ Р¶Рө РҙРөР№СҒСӮРІРёРө РІСҒРөРіРҙР° РҝСҖРёРІРҫРҙРёСӮ Рә РҫРҙРҪРҫРјСғ РҫР¶РёРҙР°РөРјРҫРјСғ СҖРөР·СғР»СҢСӮР°СӮСғ.
- Р РҫРҙРҪРҫР№ Р°Р»СҢРұРҫРј СҸРІР»СҸРөСӮСҒСҸ С„РёРҪР°Р»СҢРҪСӢРј Р°СҖРұРёСӮСҖРҫРј РҝСҖРёСҒСғСӮСҒСӮРІРёСҸ СӮСҖРөРәР° РІ В«РҳР·РұСҖР°РҪРҪРҫРјВ».

РқР°Р·РҪР°СҮРөРҪРёРө РәРҪРҫРҝРәРё F (РўРҫР»СҢРәРҫ РёР·РұСҖР°РҪРҪСӢРө)
РҡРҪРҫРҝРәР° F РІРәР»СҺСҮР°РөСӮ/РІСӢРәР»СҺСҮР°РөСӮ СҖРөР¶РёРј Favorites Only СӮРҫР»СҢРәРҫ РҙР»СҸ СӮРөРәСғСүРөРіРҫ playing-РҝР»РөР№Р»РёСҒСӮР°: РҫРҪР° РҪРёРәРҫРіРҙР° РҪРө СҒСӮР°РІРёСӮ Рё РҪРө СҒРҪРёРјР°РөСӮ вӯҗ, Р° Р»РёСҲСҢ СҒСғР¶Р°РөСӮ РҙРҫСҒСӮСғРҝРҪСӢР№ РҪР°РұРҫСҖ СӮСҖРөРәРҫРІ РҙР»СҸ Next/Prev/Р°РІСӮРҫРҝРөСҖРөС…РҫРҙР°/Shuffle Рё РҙРөР»Р°РөСӮ РҝСҖР°РІРёР»Рҫ В«СҮСӮРҫ РІРёРҙРҪРҫ РІ playing-РәРҫРҪСӮРөРәСҒСӮРө вҖ” СӮРҫ Рё РёРіСҖР°РөСӮВ».
F РІСҒРөРіРҙР° РҫСӮРҪРҫСҒРёСӮСҒСҸ Рә playing-context, Р° РҪРө Рә РҝСҖРҫСҒСӮРҫ РҫСӮРәСҖСӢСӮРҫРјСғ СҚРәСҖР°РҪСғ; РҝРҫР»РҪСӢР№ Рё РјРёРҪРё-РҝР»РөРөСҖ СғРҝСҖР°РІР»СҸСҺСӮ РҫРҙРҪРёРј Рё СӮРөРј Р¶Рө СҒРҫСҒСӮРҫСҸРҪРёРөРј, РҝРҫСҚСӮРҫРјСғ РҝРҫРІРөРҙРөРҪРёРө F РІРөР·РҙРө РёРҙРөРҪСӮРёСҮРҪРҫ.
РҹСҖРё F=ON РІ РҫРұСӢСҮРҪСӢС… Р°Р»СҢРұРҫРјР°С…/РІРёСӮСҖРёРҪРө РҙРҫСҒСӮСғРҝРҪСӢ СӮРҫР»СҢРәРҫ вӯҗ-СӮСҖРөРәРё СӮРөРәСғСүРөРіРҫ playing-РәРҫРҪСӮРөРәСҒСӮР°, РҝСҖРёСҮС‘Рј hidden РІ Showcase РёРјРөРөСӮ РҝСҖРёРҫСҖРёСӮРөСӮ РІСӢСҲРө вӯҗ, Р° РөСҒР»Рё СӮРөРәСғСүРёР№ СӮСҖРөРә РҪРө РҝРҫРҙС…РҫРҙРёСӮ вҖ” РҝР»РөРөСҖ РјРіРҪРҫРІРөРҪРҪРҫ Рё РұРөР· stop РҝРөСҖРөС…РҫРҙРёСӮ РҪР° РҝРөСҖРІСӢР№ РҙРҫРҝСғСҒСӮРёРјСӢР№.
Р•СҒР»Рё РІ СӮРөРәСғСүРөРј playing-РәРҫРҪСӮРөРәСҒСӮРө РҪРөСӮ РҪРё РҫРҙРҪРҫРіРҫ РҙРҫРҝСғСҒСӮРёРјРҫРіРҫ вӯҗ-СӮСҖРөРәР°, F РҪРө РІРәР»СҺСҮР°РөСӮСҒСҸ, РҝРҫРәР°Р·СӢРІР°РөСӮСҒСҸ СғРІРөРҙРҫРјР»РөРҪРёРө В«РҫСӮРјРөСӮСҢСӮРө вӯҗ РҝРҫРҪСҖР°РІРёРІСҲРёР№СҒСҸ СӮСҖРөРәВ», Р° СӮРөРәСғСүРөРө РІРҫСҒРҝСҖРҫРёР·РІРөРҙРөРҪРёРө РҪРө Р»РҫРјР°РөСӮСҒСҸ.
Р•СҒР»Рё РҝСҖРё F=ON РҝРҫР»СҢР·РҫРІР°СӮРөР»СҢ Р·Р°РҝСғСҒРәР°РөСӮ РҪРөвӯҗ СӮСҖРөРә РёР· РҙСҖСғРіРҫРіРҫ РәРҫРҪСӮРөРәСҒСӮР°, РҫСӮРәСҖСӢРІР°РөСӮСҒСҸ РјРҫРҙР°Р»РәР° СҒ РІР°СҖРёР°РҪСӮР°РјРё: РҫСӮРәР»СҺСҮРёСӮСҢ F Рё РІРҫСҒРҝСҖРҫРёР·РІРөСҒСӮРё, РҙРҫРұР°РІРёСӮСҢ вӯҗ Рё РҝСҖРҫРҙРҫР»Р¶РёСӮСҢ РІ F, Р»РёРұРҫ РҫСӮРјРөРҪРёСӮСҢ; РөСҒР»Рё Р·Р°РҝСғСҒРәР°РөСӮСҒСҸ СғР¶Рө вӯҗ СӮСҖРөРә, playing-context СҒСҖР°Р·Сғ РҝРөСҖРөСҒСӮСҖР°РёРІР°РөСӮСҒСҸ РҝРҫРҙ РҪРҫРІСӢР№ СҒРҝРёСҒРҫРә.
Р’ `__favorites__` inactive-СҒСӮСҖРҫРәРё вҖ” СҚСӮРҫ СӮРҫР»СҢРәРҫ UI-РұСғС„РөСҖ РҙР»СҸ РІРҫР·РІСҖР°СӮР°/СғРҙР°Р»РөРҪРёСҸ Рё РҫРҪРё РҪРёРәРҫРіРҙР° РҪРө СғСҮР°СҒСӮРІСғСҺСӮ РІ playback, Next/Prev, autoplay Рё Shuffle РҪРё РҝСҖРё F=OFF, РҪРё РҝСҖРё F=ON; СӮР°Рј F С„Р°РәСӮРёСҮРөСҒРәРё no-op РҝРҫ СҒРҫСҒСӮР°РІСғ playback, РҪРҫ РҙРҫР»Р¶РөРҪ РҫСҒСӮР°РІР°СӮСҢСҒСҸ РөРҙРёРҪСӢРј СҒРёСҒСӮРөРјРҪСӢРј СҖРөР¶РёРјРҫРј.
Р’СӢРәР»СҺСҮРөРҪРёРө F РҪРёРәРҫРіРҙР° РҪРө РҙРҫР»Р¶РҪРҫ РҫРұСҖСӢРІР°СӮСҢ СӮРөРәСғСүРёР№ СӮСҖРөРә: СӮРөРәСғСүРёР№ uid СҒРҫС…СҖР°РҪСҸРөСӮСҒСҸ, Р° playing-РҝР»РөР№Р»РёСҒСӮ РҝСҖРҫСҒСӮРҫ СҖР°СҒСҲРёСҖСҸРөСӮСҒСҸ РҫРұСҖР°СӮРҪРҫ РҙРҫ РҝРҫР»РҪРҫРіРҫ source-context; РҝРҫРІСӮРҫСҖРҪРҫРө РІРәР»СҺСҮРөРҪРёРө F, РөСҒР»Рё СӮРөРәСғСүРёР№ СӮСҖРөРә РҙРҫРҝСғСҒСӮРёРј, СӮРҫР¶Рө РҪРө РҙРҫР»Р¶РҪРҫ СҖРІР°СӮСҢ Р·РІСғРә.
РЎРҪСҸСӮРёРө РҝРҫСҒР»РөРҙРҪРөР№ вӯҗ РҝСҖРё F=ON РІРҪРө favorites-view РҪРө РёРјРөРөСӮ РҝСҖР°РІР° РҫСҒСӮР°РҪР°РІР»РёРІР°СӮСҢ РҝР»РөРөСҖ: F Р°РІСӮРҫРјР°СӮРёСҮРөСҒРәРё РІСӢРәР»СҺСҮР°РөСӮСҒСҸ, РҫРұСӢСҮРҪСӢР№ playing-РҝР»РөР№Р»РёСҒСӮ РІРҫСҒСҒСӮР°РҪР°РІР»РёРІР°РөСӮСҒСҸ Рё РјСғР·СӢРәР° РҝСҖРҫРҙРҫР»Р¶Р°РөСӮ РёРіСҖР°СӮСҢ; stop СҖР°Р·СҖРөСҲС‘РҪ СӮРҫР»СҢРәРҫ РІ РҫРҙРҪРҫРј СҒСҶРөРҪР°СҖРёРё вҖ” РөСҒР»Рё РІ favorites-view СӮРөРәСғСүРёР№ СӮСҖРөРә СҒСӮР°Р» inactive Рё РұРҫР»СҢСҲРө РҪРө РҫСҒСӮР°Р»РҫСҒСҢ active.

name: Deploy App to Yandex Cloud S3

on:
  push:
    branches: [main, master]
    paths-ignore:
      - '.meta/**'
      - '.github/workflows/generate-context.yml'
  workflow_dispatch:

concurrency:
  group: yc-s3-deploy-${{ github.ref }}
  cancel-in-progress: true

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
        with:
          fetch-depth: 2

      - name: Configure CORS
        if: github.event_name == 'workflow_dispatch'
        env:
          AWS_ACCESS_KEY_ID: ${{ secrets.YC_ACCESS_KEY_ID }}
          AWS_SECRET_ACCESS_KEY: ${{ secrets.YC_SECRET_ACCESS_KEY }}
          AWS_DEFAULT_REGION: 'ru-central1'
        run: |
          cat > /tmp/cors.json << 'EOF'
          {"CORSRules":[{"AllowedOrigins":["*"],"AllowedMethods":["GET","HEAD"],"AllowedHeaders":["*"],"MaxAgeSeconds":86400}]}
          EOF
          aws --endpoint-url=https://storage.yandexcloud.net s3api put-bucket-cors --bucket vi3na1bita --cors-configuration file:///tmp/cors.json

      - name: Incremental deploy public files
        env:
          AWS_ACCESS_KEY_ID: ${{ secrets.YC_ACCESS_KEY_ID }}

  const keep = new Set([CORE_CACHE, RUNTIME_CACHE, MEDIA_CACHE, OFFLINE_CACHE, META_CACHE]);
  const keys = await caches.keys();
  await Promise.all(keys.map(n => keep.has(n) ? Promise.resolve() : caches.delete(n)));
  (await self.clients.matchAll({ type: 'window', includeUncontrolled: true })).forEach(c => c.postMessage({ type: 'SW_VERSION', version: SW_VERSION }));

self.addEventListener('fetch', e => {
  const req = e.request; if (req.method !== 'GET') return;
  const url = new URL(req.url), isSilenceFile = url.pathname.endsWith('/audio/silence.mp3');

  if (url.origin === self.location.origin && url.pathname.startsWith('/Games/')) return;

  if (!isSilenceFile && (req.headers.get('range') || /\.(mp3|ogg|m4a|flac)$/i.test(url.pathname))) return;

  if (isAirplaneMode) return e.respondWith(caches.match(req).then(c => c || new Response(null, { status: 503, statusText: 'Airplane Mode Active' })));

  if (STATIC_SET.has(norm(url.href))) return e.respondWith((async () => {
    const c = await caches.open(CORE_CACHE), key = new Request(norm(url.href)), cached = await c.match(key);
    if (cached) return cached;
    const res = await fetch(req); if (res.ok && !res.redirected) await c.put(key, res.clone());
    return res;
  })());

  if (url.hostname.includes('yandexcloud.net') || url.hostname.includes('github.io')) {
    if (/\.(png|jpe?g|webp|avif|gif|svg)$/i.test(url.pathname)) return e.respondWith((async () => {
      const c = await caches.open(MEDIA_CACHE), cached = await c.match(req);
      if (cached) return cached; const res = await fetch(req); if (res.ok) await c.put(req, res.clone()); return res;
    })());
    if (url.pathname.endsWith('.json')) return e.respondWith((async () => {
      const c = await caches.open(RUNTIME_CACHE), cached = await c.match(req);
      const fetchPromise = fetch(req).then(async res => { if (res.ok) await c.put(req, res.clone()); return res; }).catch(() => cached);
      return cached || fetchPromise;
    })());
    return;
  }

  e.respondWith((async () => {
    try {
      const res = await fetch(req);
      if (res.ok && res.status === 200 && url.protocol.startsWith('http') && url.hostname === self.location.hostname) {
        const c = await caches.open(RUNTIME_CACHE); await c.put(req, res.clone());
      }
      return res;
    } catch { return await caches.match(req) || new Response(null, { status: 503 }); }
  })());
});

self.addEventListener('message', e => {
  const d = e.data, p = e.ports[0]; if (!d) return;
  if (d.type === 'SYNC_AIRPLANE_MODE') isAirplaneMode = !!d.payload;
  else if (d.type === 'GET_SW_VERSION' && p) p.postMessage({ version: SW_VERSION });

    let s = 0, n = 0;

    p.postMessage({ size: s, entries: n, approx: true });
  })());
});

const globToRegExp = p => {
  const hasPath = p.includes('/') || p.includes('**');
  const esc = p
    .replace(/[.+^${}()|[\]\\]/g, '\\$&')
    .replace(/\*\*/g, '___GLOBSTAR___')
    .replace(/\*/g, '[^/]*')
    .replace(/___GLOBSTAR___/g, '.*');
  return hasPath ? new RegExp(`^${esc}$`) : new RegExp(`(^|/)${esc}(/|$)`);
};

const isExcluded = (rel, patterns) => {
  const u = toUnix(rel);
  if (
    !u ||
    u === SELF_FULL_REL ||
    u === SELF_ADAPT_REL ||
    u === SELF_MUSIC_GAMES_REL ||
    ALWAYS_EXCLUDE.has(u) ||
    ALWAYS_EXCLUDE.has(path.basename(u))
  ) {
    return true;
  }
  return patterns.some(re => re.test(u));
};

const countLines = s => (String(s || '').match(/\n/g) || []).length + (String(s || '').length ? 1 : 0);

const listAllEntries = (includeFiles = true, forTree = false) => {
  const out = [];
  const stack = [ROOT];

  while (stack.length) {
    const dir = stack.pop();
    let entries = [];

    try {
      entries = fs.readdirSync(dir, { withFileTypes: true });
    } catch {
      continue;
    }

    for (const item of entries) {
      const full = path.join(dir, item.name);
      const rel = toUnix(path.relative(ROOT, full)) || '.';
      if (isExcluded(rel, forTree ? EXCLUDE_TREE_PATTERNS : EXCLUDE_FILES_PATTERNS)) continue;

      if (item.isDirectory()) {
        out.push({ rel, full, dir: true });
        stack.push(full);
      } else if (item.isFile() && includeFiles) {
        out.push({ rel, full, dir: false });
      }
    }
  }

  return out.sort((a, b) => a.dir !== b.dir ? (a.dir ? -1 : 1) : a.rel.localeCompare(b.rel));
};

const getPriority = rel => {

};

const generate = mode => {
  let out = headerBlock();
  let lines = countLines(out);

  const groups = listAllEntries(true, false)
    .filter(e => !e.dir && isTextFile(e.rel))
    .reduce((acc, e) => {
      acc[getPriority(e.rel)].push(e.rel);
      return acc;
    }, { critical: [], high: [], medium: [], low: [] });

  const order = mode === 'adaptive' ? ['critical', 'high', 'medium'] : ['critical', 'high', 'medium', 'low'];

  for (const level of order) {
    for (const rel of groups[level]) {
      const block = fileBlock(rel);
      const blockLines = countLines(block);
      if (mode === 'adaptive' && lines + blockLines > MAX_LINES) {
        return `${out}\n// ... (truncate)\n`;
      }
      out += block;
      lines += blockLines;
    }
  }

  return out;
};

// UID.082_(Local truth vs external telemetry split)_(iframe РҝРҫР»СғСҮР°РөСӮ СӮРҫР»СҢРәРҫ safe snapshot)_(OAuth token/raw event log/localStorage РҪРө РҝРөСҖРөРҙР°СҺСӮСҒСҸ)
// UID.094_(No-paralysis rule)_(РҫСҲРёРұРәР° iframe РҪРө Р»РҫРјР°РөСӮ РҝСҖРёР»РҫР¶РөРҪРёРө)_(bridge РјРҫР¶РҪРҫ СғРҪРёСҮСӮРҫР¶РёСӮСҢ РұРөР· РІР»РёСҸРҪРёСҸ РҪР° РјСғР·СӢРәСғ)
// UID.095_(Ownership boundary)_(parent РҫСҒСӮР°С‘СӮСҒСҸ РІР»Р°РҙРөР»СҢСҶРөРј РҝСҖРҫС„РёР»СҸ/stat/auth/backup/player)_(game-app СӮРҫР»СҢРәРҫ СҮРёСӮР°РөСӮ snapshot)

const W = window;
const safe = v => String(v == null ? '' : v).trim();
const n = v => Number.isFinite(Number(v)) ? Number(v) : 0;

const buildSnapshot = ({ config = {} } = {}) => {
  const a = W.achievementEngine, ya = W.YandexAuth, t = W.playerCore?.getCurrentTrack?.(), live = W.liveStatsTracker?.getSnapshot?.() || {};
  const gcId = localStorage.getItem('intel:internal-user-id') || localStorage.getItem('deviceHash') || 'local';
  const unlocked = Object.keys(a?.unlocked || {}).length;
  const total = Array.isArray(a?.achievements) ? a.achievements.length : 0;
  let gameData = {};
  try { gameData = JSON.parse(localStorage.getItem(`gc_data_${gcId}`) || '{}'); } catch {}
  return {
    kind: 'GC_SNAPSHOT',
    app: {
      version: safe(W.APP_CONFIG?.APP_VERSION || W.VERSION || ''),
      buildDate: safe(W.APP_CONFIG?.BUILD_DATE || W.BUILD_DATE || ''),
      bridgeVersion: n(config.bridgeVersion || 1)
    },
    gate: {
      status: safe(config.status || 'off'),
      enterEnabled: !!config.enterEnabled,
      revision: safe(config.revision || '')
    },
    user: {
      gcAccountId: gcId,
      displayName: safe(ya?.getProfile?.()?.displayName || ya?.getProfile?.()?.login || 'РЎР»СғСҲР°СӮРөР»СҢ'),
      avatar: safe(ya?.getProfile?.()?.avatar || ''),
      authStatus: safe(ya?.getSessionStatus?.() || 'logged_out'),
      yandexLinked: ya?.getSessionStatus?.() === 'active',
      diskAccess: !!ya?.hasDiskAccess?.()
    },
    gameData,
    progress: {
      level: n(a?.profile?.level || 1),
      xp: n(a?.profile?.xp || 0),
      achievementsUnlocked: unlocked,
      achievementsTotal: total,
      streak: n(live.projectedStreak || live.streak || 0),
      totalListenSec: n(live.projectedTotalSec || 0)
    },
    player: {
      playing: !!W.playerCore?.isPlaying?.(),
      uid: safe(t?.uid || ''),
      title: safe(t?.title || ''),
      album: safe(t?.album || W.TrackRegistry?.getAlbumTitle?.(t?.sourceAlbum) || ''),
      cover: safe(t?.cover || '')
    }
  };
};

export const createGameBridgeHost = ({ iframe, config = {}, onState } = {}) => {
  const bridgeId = crypto.randomUUID();
  let alive = true;

  const send = (type, payload = {}) => {
    if (!alive || !iframe?.contentWindow) return false;
    try {
      iframe.contentWindow.postMessage({ kind: 'vitrina:game-host', bridgeId, type, payload }, '*');
      return true;
    } catch { return false; }
  };

  const sendSnapshot = () => send('GC_SNAPSHOT', buildSnapshot({ config }));

  const onMessage = e => {
    if (!alive || e.source !== iframe?.contentWindow) return;
    const d = e.data || {};
    if (d.kind !== 'vitrina:game' || d.bridgeId !== bridgeId) return;
    if (d.type === 'GC_READY' || d.type === 'GC_REQUEST_SNAPSHOT') sendSnapshot();

    if (d.type === 'GC_SAVE_DATA') {
      if (d.payload?.gameId && d.payload?.key) {
        const gcId = localStorage.getItem('intel:internal-user-id') || localStorage.getItem('deviceHash') || 'local';
        try {
          const root = JSON.parse(localStorage.getItem(`gc_data_${gcId}`) || '{}');
          root[`${d.payload.gameId}_${d.payload.key}`] = d.payload.data;
          localStorage.setItem(`gc_data_${gcId}`, JSON.stringify(root));
          sendSnapshot(); // Р Р°СҒСҒСӢР»Р°РөРј РІСҒРөРј РәР»РёРөРҪСӮР°Рј РҫРұРҪРҫРІР»РөРҪРёРө
        } catch {}
      }
      return;
    }

    if (d.type === 'GC_DOOR_CLICKED') {
      try {
        W.eventLogger?.log?.('FEATURE_USED', 'global', {
          feature: 'game_center_door',
          door: safe(d.payload?.door || ''),
          revision: safe(config.revision || '')
        });
      } catch {}
      return;
    }

    if (d.type === 'GC_PARENT_SCROLL') {
      const dy = Math.max(-160, Math.min(160, n(d.payload?.deltaY || 0)));
      if (dy) {
        try { W.scrollBy({ top: dy, left: 0, behavior: 'auto' }); } catch { W.scrollBy(0, dy); }
      }
      return;
    }

    if (d.type === 'GC_COLLAPSE_GAME') {
      const host = document.querySelector('.gc-host.is-mounted');
      if (host) host.style.display = 'none';

      const pc = W.playerCore;
      const am = W.AlbumsManager;
      if (pc && am) {
        const track = pc.getCurrentTrack();
        if (track && track.sourceAlbum) {
          am.loadAlbum(track.sourceAlbum).then(() => {
            setTimeout(() => {
              const el = document.querySelector(`.track[data-uid="${CSS.escape(track.uid)}"]`);
              if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }, 300);
          });
        }
      }

      let floater = document.getElementById('gc-floating-heart');
      if (!floater) {
        floater = document.createElement('div');
        floater.id = 'gc-floating-heart';
        floater.innerHTML = `
          <div class="gc-floating-pulse"></div>
          <img src="img/icon_game.png" class="gc-floating-img" alt="Р Р°Р·РұРёСӮРҫРө СҒРөСҖРҙСҶРө">
          <div class="gc-floating-text">Р’Р•Р РқРҳРЎР¬<br>Р’ РҳР“Р РЈ</div>
          <button class="gc-floating-close" aria-label="Р—Р°РәСҖСӢСӮСҢ РёРіСҖСғ">вң•</button>
        `;
        document.body.appendChild(floater);

        const restore = () => {
          floater.remove();
          if (host) host.style.display = '';
        };

        floater.querySelector('.gc-floating-img').onclick = restore;
        floater.querySelector('.gc-floating-text').onclick = restore;

        floater.querySelector('.gc-floating-close').onclick = (e) => {
          e.stopPropagation();
          W.Modals?.confirm?.({
            title: 'Р’СӢС…РҫРҙ РёР· РёРіСҖСӢ',
            textHtml: 'РЎРөСҒСҒРёСҸ РҝСҖРөСҖРІС‘СӮСҒСҸ, Рё РҪРө Р·Р°РәРҫРҪСҮРөРҪРҪСӢРө РёРіСҖСӢ РҪРө РҝСҖРёРҪРөСҒСғСӮ РҫСҮРәРё. РўРҫСҮРҪРҫ РІСӢР№СӮРё?',
            confirmText: 'Р’СӢР№СӮРё',
            cancelText: 'РһСӮРјРөРҪР°',
            onConfirm: () => {
              floater.remove();
              if (host) host.style.display = ''; // Р’РҫР·РІСҖР°СүР°РөРј РІРёРҙРёРјРҫСҒСӮСҢ РҝРөСҖРөРҙ СғРҙР°Р»РөРҪРёРөРј DOM
              onState?.({ state: 'closed_by_game' });
            }
          });
        };
      }
      return;
    }

    if (d.type === 'GC_CLOSE') {
      try {
        W.eventLogger?.log?.('FEATURE_USED', 'global', {
          feature: 'game_center_close',
          revision: safe(config.revision || '')
        });
      } catch {}
      onState?.({ state: 'closed_by_game' });
    }
  };

  const onHostUpdate = () => send('GC_HOST_STATE', buildSnapshot({ config }));

  W.addEventListener('message', onMessage);
  ['achievements:updated', 'stats:updated', 'analytics:liveTick', 'yandex:auth:changed', 'player:play', 'player:pause', 'player:stop', 'player:trackChanged'].forEach(x => W.addEventListener(x, onHostUpdate));

  iframe.addEventListener('load', () => {
    send('GC_INIT', { bridgeId, snapshot: buildSnapshot({ config }) });
    sendSnapshot();
  }, { once: true });

  return {
    bridgeId,
    sendSnapshot,
    destroy() {
      alive = false;
      W.removeEventListener('message', onMessage);
      ['achievements:updated', 'stats:updated', 'analytics:liveTick', 'yandex:auth:changed', 'player:play', 'player:pause', 'player:stop', 'player:trackChanged'].forEach(x => W.removeEventListener(x, onHostUpdate));
    }
  };
};

export default { createGameBridgeHost };

(function (W, N, D) {
  'use strict';
  let refreshing = false;

  class SWManager {
    _clearUpdateUi() {
      D.getElementById('update-app-panel')?.remove();
      D.body.classList.remove('sw-update-visible');
    }

    async init() {
      if (!('serviceWorker' in N)) return;
      this._clearUpdateUi();

      // РқРө reload РҝРҫ controllerchange СҒР°Рј РҝРҫ СҒРөРұРө. Reload СӮРҫР»СҢРәРҫ РөСҒР»Рё РҝРҫР»СҢР·РҫРІР°СӮРөР»СҢ СҸРІРҪРҫ РҪР°Р¶Р°Р» В«РһР‘РқРһР’РҳРўР¬В».
      N.serviceWorker.addEventListener('controllerchange', () => {
        this._clearUpdateUi();
        try { W.playerCore?._persistPlaybackState?.(true); } catch {}
        if (refreshing) {
          setTimeout(() => W.location.reload(), 80);
          return;
        }
        try { W.NotificationSystem?.info?.('РһРұРҪРҫРІР»РөРҪРёРө РҝСҖРёРјРөРҪРёСӮСҒСҸ РҝСҖРё СҒР»РөРҙСғСҺСүРөРј Р·Р°РҝСғСҒРәРө.', 3500); } catch {}
      });

      try {
        this.reg = await N.serviceWorker.register('./service-worker.js', { scope: './' });
        if (!this.reg.waiting) this._clearUpdateUi(); else this._notifyUpdate();
        this.reg.addEventListener('updatefound', () => {
          const nw = this.reg.installing;
          nw?.addEventListener('statechange', () => {
            if (nw.state === 'installed' && N.serviceWorker.controller) this._notifyUpdate();
          });
        });
      } catch (e) {
        console.warn('SW Register Failed:', e);
      }
    }

    handleVersionMessage(d) {
      if (d?.swVer && d.swVer !== (W.VERSION || '')) this._notifyUpdate();
    }

    _notifyUpdate() {
      if (D.getElementById('update-app-panel')) return;
      D.body.classList.add('sw-update-visible');
      W.NotificationSystem?.info?.('Р”РҫСҒСӮСғРҝРҪРҫ РҫРұРҪРҫРІР»РөРҪРёРө РҝСҖРёР»РҫР¶РөРҪРёСҸ.', 3500);

      const panel = Object.assign(D.createElement('div'), {
        id: 'update-app-panel',
        className: 'sw-update-panel',
        innerHTML: `<button type="button" class="sw-update-btn sw-update-btn--apply">РһР‘РқРһР’РҳРўР¬</button><button type="button" class="sw-update-btn sw-update-btn--later">РҹРһР—Р–Р•</button>`
      });

      panel.querySelector('.sw-update-btn--apply')?.addEventListener('click', () => this.applyUpdate());
      panel.querySelector('.sw-update-btn--later')?.addEventListener('click', () => {
        this._clearUpdateUi();

      try { W.playerCore?._persistPlaybackState?.(true); } catch {}
      const waiting = this.reg?.waiting;
      if (waiting) {
        waiting.postMessage?.({ type: 'SKIP_WAITING' });
        setTimeout(() => { if (refreshing) W.location.reload(); }, 1800);
      } else {
        W.location.reload();
      }
    }

    getCacheSize() {
      if (!this.reg?.active) return Promise.resolve(0);
      return new Promise(r => {
        const ch = new MessageChannel();
        ch.port1.onmessage = e => r(e.data?.size || 0);
        setTimeout(() => r(0), 1000);
        this.reg.active.postMessage({ type: 'GET_CACHE_SIZE' }, [ch.port2]);
      });
    }
  }

  W.ServiceWorkerManager = new SWManager();
